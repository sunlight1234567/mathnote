\chapter{积分等式与积分不等式}
% ===================================
这一部分主要是使用前面学到的定理来解决有关积分的等式、不等式问题。

难度较大，做题的时候注意心态不要崩。


\begin{induction}
    [由中值定理推导的积分等式]
    设$f(x),g(x)$在$[a,b]$上连续，且$g(x)$不变号，则存在$\xi\in[a,b]$使得
    \[
        \int_a^b f(x)g(x)\mathrm{d}x = f(\xi)\int_a^b g(x)\mathrm{d}x.
    \]
\end{induction}
\begin{proof}
    设$g(x)\geqslant 0$，则存在$\xi_1,\xi_2\in[a,b]$使得
    \[
        f(\xi_1) = \min_{x\in[a,b]}f(x),\quad f(\xi_2) = \max_{x\in[a,b]}f(x).
    \]
    则
    \[
        f(\xi_1)\int_a^b g(x)\mathrm{d}x \leqslant \int_a^b f(x)g(x)\mathrm{d}x \leqslant f(\xi_2)\int_a^b g(x)\mathrm{d}x.
    \]
    由介值定理，存在$\xi\in[a,b]$使得
    \[
        \int_a^b f(x)g(x)\mathrm{d}x = f(\xi)\int_a^b g(x)\mathrm{d}x.
    \]
    当$g(x)\leqslant 0$时同理。
\end{proof}

\begin{induction}
    [使用夹逼准则求积分的极限]
    设$f(x),g(x),h(x)$在$[a,b]$上连续，
    且满足$f(x)\leqslant g(x)\leqslant h(x)$。

    若$\int_a^b f(x)\mathrm{d}x = \int_a^b h(x)\mathrm{d}x$，
    则$\lim\int_a^b g(x)\mathrm{d}x = \lim \int_a^b f(x)\mathrm{d}x$。
\end{induction}

\begin{induction}
    [$\bigstar$积分不等式的证明方法]
    (1)利用积分函数的单调性;
    
    (2)常量变量化：将某一个积分限变量化，构造变限积分函数，利用单调性证明该函数的单调性;
    
    (3)利用积分中值定理。    
    
    (4)利用泰勒公式将被积函数展开。
\end{induction}

\begin{example}
    [常量变量化方法]
    设$f(x)$在$[a,b]$上连续且$f(x)$严格单调增加，证明：
    $$(a+b)\int_{a}^{b}f(x)\mathrm{d}x < 2\int_{a}^{b}xf(x)\mathrm{d}x\mathstrut$$

    提示：将b变量化构造辅助函数$F(x)=(a+x)\int_{a}^{b}f(t)\mathrm{d}t - 2\int_{a}^{x}tf(t)\mathrm{d}t$
\end{example}

\begin{example}
    [积分中值定理方法]
    设$f'(x)$在$[a,b]$上连续$f(0)=0$，证明：
    $$|\int_{a}^{b}f(x)dx| \leq \dfrac{Ma^2}{2}$$
    其中$M=\max_{x\in[a,b]}|f'(x)|$。
\end{example}

\begin{example}
    [泰勒公式方法]
    设$f(x)$在$[0,1]$上二阶可导，且$f(\dfrac{1}{2})=1,f^{''}(x) > 0$，证明：
    $\int_{0}^{1}f(x)\mathrm{d}x\geq 1$
\end{example}
\begin{note}
    在$x_0=\dfrac{1}{2}$处泰勒展开，带佩亚诺余项，奇次项的积分为0.
\end{note}
