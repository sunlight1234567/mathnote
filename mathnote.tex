\section{极限的性质}

\begin{definition}[函数极限的 $\varepsilon$-$\delta$ 定义]
设函数 $f(x)$ 在点 $x_0$ 的某去心邻域内有定义. 若存在常数 $A$, 对于任意给定的 $\varepsilon > 0$, 总存在 $\delta > 0$, 使得当 $0 < |x - x_0| < \delta$ 时, 有
\[
    |f(x) - A| < \varepsilon,
\]
则称 $A$ 为函数 $f(x)$ 当 $x \to x_0$ 时的\textbf{极限}, 记作 $\lim_{x \to x_0} f(x) = A$.
\end{definition}

\begin{theorem}[极限的性质]
\textbf{唯一性}：若 $\lim_{x \to x_0} f(x)$ 存在, 则该极限值唯一.

\textbf{局部有界性}：若 $\lim_{x \to x_0} f(x)$ 存在, 则存在 $x_0$ 的某去心邻域, 在该邻域内 $f(x)$ 有界.

\textbf{局部保号性}：若 $\lim_{x \to x_0} f(x) = A \neq 0$, 则存在 $\delta > 0$, 使得当 $0 < |x - x_0| < \delta$ 时, $f(x) > \dfrac{A}{2} > 0$.
\end{theorem}

\begin{note}
保号性的逆命题\textbf{不成立}: 若在 $x_0$ 的去心邻域内 $f(x) > 0$, 只能推出 $\lim_{x \to x_0} f(x) \ge 0$, 不能保证严格大于零. 例如 $f(x) = x^2$, $x_0 = 0$.
\end{note}


\section{极限的计算}
\begin{induction}[极限的计算方法]
\textbf{四则运算}：极限的四则运算与数列极限类似, 但要注意函数极限的定义域问题.

\textbf{洛必达法则}：使用柯西中值定理证明，注意其条件的充分性

\textbf{泰勒公式}：当 $x \to 0$ 时, 若 $f(x)$ 在 $x = 0$ 处具有 $n$ 阶可导数, 且 $f(0) = f'(0) = \cdots = f^{(n-1)}(0) = 0$, 则
\[
    f(x) = f(0) + f'(0)x + \dfrac{f^{(2)}(0)}{2!}x^{2} + \cdots + \dfrac{f^{(n-1)}(0)}{(n-1)!}x^{n-1} + R_n(x),
\]
其中 $R_n(x)$ 可以为拉格朗日余项、佩亚诺余项.

\textbf{夹逼准则}：适当使用放缩

\textbf{两个重要极限}: $\lim_{x \to 0} \dfrac{\sin x}{x} = 1$ 和 $\lim_{x \to \infty}\left(1 + \dfrac{1}{x}\right)^{x} = e$.

$\star$ \textbf{第二重要极限的推论}: $\lim_{u\to1} u^v = \lim_{u\to1}\{[1+(u-1)]^{1/(u-1)}\}^{(u-1)v} = e^{\lim_{u\to1}(u-1)v}.$

\textbf{无穷小替换}：适当使用无穷小替换, 例如 $\ln(x + \sqrt{1+x^2}) \sim x$.
\end{induction}

\begin{note}
极限的计算方法并非相互独立, 例如洛必达法则的证明就依赖于柯西中值定理, 泰勒公式的证明也依赖于洛必达法则.

在求幂指式(如$u^v$)的极限时, 可以将其转化为指数幂的形式 $e^{v\ln u}$, 或者凑第二重要极限$\left(1 + \dfrac{1}{x}\right)^{x}$.
\end{note}


\section{函数的连续性}

\begin{induction}[函数连续性]
连续的定义：\textbf{左右极限相等且等于函数值}.

$\Rightarrow $连续的函数在定义域内的每一点都具有极限, 且极限值等于函数值.

连续函数的四则运算、复合函数、反函数（若存在）也连续.

$\star$ \textbf{连续的局部保号性}：与极限的保号性一致
\end{induction}

\textbf{第二类间断点}：无穷间断点，震荡间断点$(\sin \frac{1}{x}, x\to 0)$.
\end{induction}

\begin{induction}[数列求和方法]
常见的数列：等差，等比数列，差比数列

常见数列的和：

\end{induction}

\begin{definition}[收敛数列]
设数列 $\{a_n\}$ 满足: 对于任意给定的 $\varepsilon > 0$, 总存在 $N \in \mathbb{N}^+$, 使得当 $n > N$ 时, 有
\[
    |a_n - A| < \varepsilon,
\]
则称数列 $\{a_n\}$ 收敛于 $A$, 记作 $\lim_{n \to \infty} a_n = A$.

函数的极限是局部的性质，而数列的极限是全局的性质，\textbf{与前有限项无关}.

数列极限的性质与函数极限类似, 包括\textbf{唯一性}、\textbf{局部有界性}、\textbf{局部保号性}等.

数列的极限等于其对应函数在正无穷远处的极限, 即 $\lim_{n \to \infty} a_n = \lim_{x \to \infty} f(x)$, 其中 $f(n) = a_n$.

\end{definition}


\begin{corollary}[数列${a_n}$收敛于$A$]
$\rightarrow$  其任一子列也收敛于$A$.

$\rightarrow$  ${|a_n|}$也收敛于$A$.
\end{corollary}


\begin{theorem}[海涅定理/归结定理]
$f(x)$ 在$x_0$附近有定义，$\lim_{x \to x_0}f(x) = A \Leftrightarrow$ 以$x_0$为极限的数列${a_n}$，极限$\lim_{n\to\infty} f(a_n) = A$.

其本质为复合函数的极限，$n\to\infty, a_n \to x_0 \rightarrow f(a_n) \rightarrow A$, 注意该极限与$f(x_0)$的值无关.
\end{theorem}


\begin{theorem}[夹逼准则]
从某一项开始满足两边夹即可

常用的不等式放缩：$P_{133}$

$$|a\pm b| \le |a| + |b|$$

$$||a|-|b|| \le |a-b|$$

$$\sqrt{ab} \le \frac{a+b}{2} \le \sqrt{\frac{a^2+b^2}{2}}$$

$$
\sqrt[3]{abc}\le \frac{a+b+c}{3}\le \sqrt{\frac{a^2+b^2+c^2}{3}}.
$$
\textbf{结论：}$\lim_{n \to \infty} \sqrt[n]{a_1^n + a_2^n + \cdots + a_k^n} = \max\{a_1, a_2, \ldots, a_k\}$
\end{theorem}

\begin{corollary}[压缩映射原理一]
设数列${a_n}$满足：存在常数$0<q<1$，使得当$n$充分大时，$|a_{n+1}-a|\le q|a_n-a|$, 则数列${a_n}$收敛于$a$.
\end{corollary}
\begin{proof}
证明：$|a_{n+1}-a|\le q|a_n-a|\le q^2|a_{n-1}-a|\le \cdots \le q^{n}|a_1-a|$. 从而数列${a_n}$满足夹逼准则，收敛.
\end{proof}
\begin{note}
由于$0<q<1$, $a_n$ 到 $a$ 的距离以指数级的速度收敛到0, 因此该定理也被称为\textbf{指数收敛定理}.(该名称由AI给出)

还有压缩映射原理二。在使用时均需证明。
\end{note}   


\begin{theorem}[单调有界定理]
单调递增且有上界的数列必收敛于其上确界; 单调递减且有下界的数列必收敛于其下确界.

证明数列的单调性：\textbf{做差法}, \textbf{做商法}, \textbf{数学归纳法}.
\end{theorem}                

\begin{example}
$x_1 = 2, x_n+(X_n - 4)x_{n-1}=3(n=2,3,4,\ldots)$,证明数列${x_n}$极限存在，并求出该极限.
\end{example}

\section{导数}
\begin{definition}[导数的定义]
设函数 $y = f(x)$ 在点 $x_0$ 的某邻域内有定义, 若极限
\[
    \lim_{\Delta x \to 0} \dfrac{f(x_0 + \Delta x) - f(x_0)}{\Delta x}
\]
存在, 则称 $f(x)$ 在点 $x_0$ 处\textbf{可导}, 并称此极限值为 $f(x)$ 在 $x_0$ 处的导数, 记作 $f'(x_0)$.
等价形式：
\[
    f'(x_0) = \lim_{x \to x_0} \dfrac{f(x) - f(x_0)}{x - x_0}.
\]

由于极限存在的充要条件为左右极限均存在且相等，因此导数$f'(x_0)$存在等价于：
\[
    \lim_{x \to x_0^-} \dfrac{f(x) - f(x_0)}{x - x_0} = \lim_{x \to x_0^+} \dfrac{f(x) - f(x_0)}{x - x_0}.
\]
\[
    f_-'(x_0) = f_+'(x_0).
\]
\end{definition} 

\begin{note}
\textbf{可导必连续}：导数的定义要求函数在该点附近有定义，且极限存在，这自然蕴含了函数在该点的连续性。

\textbf{导数的增广}：
\[
    \lim_{(A-B) \to 0} \dfrac{f(x_0 + A) - f(x_0 + B)}{A - B} = \lim_{(A-B) \to 0} \dfrac{f(x_0+B+A-B)-f(x_0+B)}{A-B}=f'(x_0+B).
\]

$\star$ \textbf{定义求某一点的导数}: 当函数不具备"导数存在"的条件时，往往使用定义来求在某一点的导数.

$\star$ \textbf{凑导数来求极限}: 某些特定的极限可以通过凑导数来求解，尤其是当$f(x_0)=0$时十分好用.
\end{note}

\begin{corollary}
\textbf{求导改变函数的奇偶性，而周期保持不变}：如果$f(x)$是偶函数，则$f'(x)$是奇函数；如果$f(x)$是奇函数，则$f'(x)$是偶函数。
\end{corollary}

\begin{corollary}
$f(x)$在$x=a$连续，$F(x)=f(x)|x-a|$, 则$f(a)=0$是$F(x)$在$x=a$处可导的\textbf{充要条件}。
\end{corollary}

\begin{note}
例：求$f(x)=(x^2-1)|x^3+x^2-2x-2|$的不可导点.
（提示：将$f(x)$因式分解。）
\end{note}


\section{导数的几何意义}

\begin{induction}[导数的几何意义]
\textbf{切线斜率}：函数 $f(x)$ 在点 $x_0$ 处的导数 $f'(x_0)$ 表示函数图像在该点的切线斜率.
\end{induction}
\begin{note}
导数存在，切线一定存在；但反之不成立。如：$f(x) = \sqrt[3]{x}$在$x=0$处切线存在，但导数不存在。
\end{note}


\section{高阶导数与微分}

\begin{induction}[高阶导数]
高阶导数是对低一阶导数求导，符合一阶导数对于原函数的所有性质。

一阶倒数用于判断原函数的增减性，二阶导数用于判断一阶导数的增减性和原函数的凹凸性。
\end{induction}

\begin{definition}[可微]
在$x_0$的某一邻域内函数增量：
\[
    \Delta y = f(x_0 + \Delta x) - f(x_0) = A \Delta x + o(\Delta x), 
\]
其中$A$为常数，称函数$f(x)$在$x_0$处\textbf{可微}，并取其线性主部$A\Delta x$为函数$f(x)$在$x_0$处的微分$dy$,
记作
\[dy|_{x=x_0}=A\Delta x\text{或}df|_{x=x_0}=f'(x_0)dx.\]

\textbf{可微与可导等价}：在$x_0$处可微当且仅当在该点处存在切线，即导数存在。

\end{definition}

\begin{induction}[复合函数求导]
\[
    \text{微分不变性:}\; d(f(g(x))) = f'(g(x)) \cdot d(g(x)) = f'(g(x))g'(x)\,dx.
\]
\[
    \text{链式法则:} (f(g(x)))' = f'(g(x))g'(x),\text{或者}\dfrac{\mathrm{d}f(g(x))}{\mathrm{d}x} = \dfrac{\mathrm{d}f(g(x))}{\mathrm{d}g(x)} \cdot \dfrac{\mathrm{d}g(x)}{\mathrm{d}x}.
\]
\end{induction}

\begin{note}

\end{note}

\textbf{幂指数求导法}：当函数具有幂指式的形式$u(x)^{v(x)}$时，可以将其转化为指数幂的形式 $e^{v(x)\ln u(x)}$,然后再求导.
\end{induction}


\begin{induction}
    [高阶导数的求法]
(1)\textbf{归纳法}：逐阶求导，归纳凑通式，可以借助数学归纳法;

(2)\textbf{莱布尼兹公式}:

设$u(x),v(x)$具有$n$阶导数，则
\[(u\cdot v)^{(n)} = \sum_{k=0}^{n} \binom{n}{k} u^{(k)} \cdot v^{(n-k)}.\]
右边的形式与二项式定理类似。

当$u(x)$或$v(x)$是$x$的多项式时，右边展开就只有很少的前几项，后面均为0.

(3)\textbf{泰勒展开式}:

$f(x) = \sum_{n=0}^{\infty}\dfrac{f^{(n)}(x_0)}{n!}(x-x_0)^n$

根据已知公式将函数$f(x)$展开为幂级数，由泰勒展开式的唯一性，将其各项系数与上式各项系数一一对应(将幂级数的通项换为$x^n$便于比较)，即可求出各阶导数.
\end{induction}

    \item $(\sin ax + b)^{(n)} = a^n\sin(ax + \frac{n\pi}{2})$
    \item $(\cos ax + b)^{(n)} = a^n\cos(ax + \frac{n\pi}{2})$

    \item $(\ln(ax+b))^{(n)} = \dfrac{(-1)^{n-1} (n-1)! a^n}{(ax+b)^n}$
    \item $(e^{ax})^{(n)} = a^n e^{ax}$
\end{itemize}
\end{induction}

\section{单调性和极值}

\begin{definition}
    [极值]
$f(x)$在$x_0$某一领域有定义,邻域内任意一点$x$,均有
\[
    f(x) \le f(x_0) \quad (\text{或} f(x) \ge f(x_0)),
\]
则称$f(x)$在$x_0$处取得\textbf{极大值}（或\textbf{极小值}）.
\end{definition}
\begin{note}
    (1)间断点也可以是极值点，如
    $f(x)=\begin{cases}
        1, & x = 0, \\
        0, & x \neq 0.
    \end{cases}$在$x=0$处取得极大值.
    (2)常函数处处为极值点
\end{note}

\begin{induction}
    [单调性和极值的判断方法]
    \textbf{单调性}：区间$(a,b)$内$f'(x)\geq0$,等号仅在有限个点成立,则$f(x)$在$(a,b)$内单调递增;递减同理.
    \textbf{极值}：
    
    (1)必要条件：如果$f(x)$在$x_0$处取得极值, 则$f'(x_0) = 0$（Fermat 引理）.

    (2)第一充分条件：如果$f'(x)$在$x_0$的某邻域内由正变负, 则$f(x)$在$x_0$处取得极大值; 由负变正则取得极小值.

    (3)第二充分条件：如果$f'(x_0) = 0$且$f''(x_0)$存在, 则
    \begin{itemize}
        \item 若$f''(x_0) > 0$, 则$f(x)$在$x_0$处取得极小值;
        \item 若$f''(x_0) < 0$, 则$f(x)$在$x_0$处取得极大值;
        \item 若$f''(x_0) = 0$, 则需进一步判断.
    \end{itemize}
\end{induction}
\begin{note}
    $f'(x_0)=0$不代表$f(x)$在$x_0$处不变化，而是变化的幅度很小，如$f(x)=x^3$在$x=0$处.
    
    \textbf{判断极值的第三充分条件}：$f(x)$在$x=x_0 n$阶可导，前$n-1$阶导数在$x_0$处为0，且第$n$阶导数在$x_0$处不为0，则

    当n为偶数，$f^{(n)}(x)<0$,则$f(x)$在$x_0$处取得极大值; 当n为奇数，$f^{(n)}(x)>0$, 则$f(x)$在$x_0$处取得极小值.
\end{note}


\section{凹凸性和拐点}

\begin{definition}
    [凹凸性和拐点]
    \textbf{第一定义}：设函数$f(x)$在区间$I$上连续,恒有图像上两点间的弧在函数图像之上，则称图形是凹的，即：
    \[
    \dfrac{f(x_1) + f(x_2)}{2} > f\left(\dfrac{x_1 + x_2}{2}\right), \quad \forall x_1, x_2 \in I, x_1 \ne x_2.
    \]
    \textbf{第二定义}：设函数$f(x)$在区间$I$上连续,区间上任意一点处的切线在函数图像之下，则称图形是凹的,表达式略.    
    
    函数凹凸性改变的点叫做"拐点".
\end{definition}
\begin{note}
    \textbf{拐点是图像上的点，写$(x_0,f(x_0))$},函数在该点必须连续.
   
    凹凸性的第一定义和第二定义是等价的，且都可以推广到多元函数的情况。

    "凹"与"凸"是从函数轴正方向向下看的结果.
\end{note}

\begin{induction}
    [凹凸性和拐点的判定]
    \begin{itemize}
        \item 若$f''(x) > 0$对$I$内的任意$x$成立, 则称$f(x)$在$I$内\textbf{凹}（或\textbf{凸}）;
        \item 若$f''(x) < 0$对$I$内的任意$x$成立, 则称$f(x)$在$I$内\textbf{凸}（或\textbf{凹}）;
        \item \textbf{判断拐点的第一充分条件}：若$f''(x_0) = 0$, 且在$x_0$的某邻域内$f''(x)$由正变负或由负变正, 则称$(x_0, f(x_0))$为函数图像的一个\textbf{拐点}.
        \item \textbf{判断拐点的第二充分条件}：若$f''(x_0)=0, f'''(x_0) \neq 0$, 则$(x_0, f(x_0))$为函数图像的一个\textbf{拐点}.
    \end{itemize}
\end{induction}
\begin{note}
    (1)二阶导数不存在的点也可能是拐点，如$f(x)=\sqrt[3]{x}$在$x=0$处.

    (2)$f(x),f'(x),f''(x)$连续时,$f(x)$的拐点就是$f'(x)$的极值点.
\end{note}

\begin{corollary}
    曲线的可导点不可同时为极值点与拐点；不可导点可以同时为极值点和拐点.

    多项式函数$f(x) = (x-a)^ng(x)(n>1,g(a)\neq 0)$,$n$为偶数, $x=a$是$f(x)$的极值点；$n$为奇数, $(a,0)$是$f(x)$的拐点.
    
    例如：$f_1(x) = (x-a)^2(x^2+1), f_2(x) = (x-a)^3(x^2+1)$
\end{corollary}

\section{渐近线}
\begin{definition}
    [三种渐近线]
    \textbf{铅直渐近线}：如果$\lim_{x \to a^-} f(x) = \pm \infty$或$\lim_{x \to a^+} f(x) = \pm \infty$, 则称$x=a$是函数$f(x)$的一个铅直渐近线.

    \textbf{水平渐近线}：如果$\lim_{x \to +\infty} f(x) = b$或$\lim_{x \to -\infty} f(x) = b$, 则称$y=b$是函数$f(x)$的一个水平渐近线.

    \textbf{斜渐近线}：如果$\lim_{x \to +/-\infty} \dfrac{f(x)}{x} = k$且$\lim_{x \to +/-\infty} (f(x) - kx) = b$, 则称$y = kx + b$是函数$f(x)$的一个斜渐近线.
\end{definition}

\begin{induction}

    \]

    (4)\textbf{零点定理}
\end{induction}
\begin{note}

\end{note}


\section{导数的中值定理}
\begin{induction}[导数的中值定理]
    (1)\textbf{费马定理}：如果函数$f(x)$在$x_0$处取得极值，且$f'(x_0)$存在，则$f'(x_0)=0$.

    (2)\textbf{导数零点定理}：如果函数$f(x)$在$[a,b]$上连续，在$(a,b)$内可导，当$f_+'(a)\cdot f'(b) < 0$时，则
    \[
        \text{存在}\xi\in(a,b)\text{使得}f'(\xi)=0.
    \]
    (3)$\bigstar$ \textbf{罗尔定理}：如果函数$f(x)$在$[a,b]$上连续，在$(a,b)$内可导，并且$f(a)=f(b)$，则存在$\xi\in(a,b)$使得$f'(\xi)=0$.

    推广：如果函数$f(x)$在$(a,b)$内可导，并且$f(a)=f(b)=A$，则存在$\xi\in(a,b)$使得$f'(\xi)=0$，其中$(a,b)$可为有限区间或无穷区间，A可为有限数字，也可为无穷大.

    (4)$\bigstar$ \textbf{拉格朗日定理}：如果函数$f(x)$在$[a,b]$上连续，在$(a,b)$内可导，则存在$\xi\in(a,b)$使得
    \[
        f'(\xi) = \dfrac{f(b) - f(a)}{b - a}.
    \]
    (5)$\bigstar$ \textbf{柯西定理}：如果函数$f(x), g(x)$在$[a,b]$上连续，在$(a,b)$内可导，并且$g'(x)\neq 0$，则存在$\xi\in(a,b)$使得
    \[
        \dfrac{f'(\xi)}{g'(\xi)} = \dfrac{f(b) - f(a)}{g(b) - g(a)}.
    \]
    (6)$\bigstar$ \textbf{泰勒公式}：如果函数$f(x)$在$x_0$某个邻域内具有$n+1$阶导数，则对于任意$x\in\cup(x_0-\delta, x_0+\delta)$,$x_0,x$间存在$\xi$使得
    \[
        f(x) = f(x_0) + f'(x_0)(x-x_0) + \cdots + \dfrac{f^{(n)}(x_0)}{n!}(x-x_0)^n + \dfrac{f^{(n+1)}(\xi)}{(n+1)!}(x-x_0)^{n+1}.
    \]
    此为拉格朗日余项，佩亚诺余项为$o(x-x_0)^{n}$
\end{induction}

\section{不定积分}
\begin{definition}
    [原函数与不定积分]
    若在区间$I$上任意一点都有$F'(x) = f(x)$, 则称$F(x)$是$f(x)$的一个\textbf{原函数}, 记作$F(x) = \int f(x) dx$.
\end{definition}
\begin{note}
    $\bigstar$ 原函数不唯一，$F(x) + C$也是$f(x)$的一个原函数，其中$C$为任意常数.求不定积分时必须要加上常数$C$.

    由$F'(x) = f(x)$可知：\textbf{$F(x)$必须连续}.

    谈到原函数与不定积分，必须要指明$f(x)$的定义域，否则可能会出现不同的原函数，如$f(x)=\dfrac{1}{x}$在$(-\infty,0)$和$(0,+\infty)$上分别有不同的原函数.尽管二者可以使用$\ln|x|$表示，但定义域不同.
\end{note}

\begin{theorem}
    [原函数(不定积分)存在定理]
    (1)连续函数$f(x)$必定存在原函数$F(x)$.

    (2)含有\textbf{第一类间断点}和\textbf{无穷间断点}的函数$f(x)$在包含改间断点的区间内没有原函数.

    例如$f(x)=\dfrac{1}{x}$在$x=0$处有无穷间断点，所以在包含$x=0$的任何区间内都没有原函数.
\end{theorem}
\begin{note}
    证明见P244.

    简单来说就是：不定积分的原函数$F(x)$要求在积分区间上处处可导，
    尽管含有\textbf{第一类间断点}和\textbf{无穷间断点}的函数$f(x)$的变上限积分存在，
    但该变上限积分在间断点处不可导，所以不满足原函数的定义.
\end{note}


\section{定积分}
\begin{definition}
    [定积分]
    设函数$f(x)$在$[a,b]$上有定义, 将区间$[a,b]$等分成$n$份, 每份长度为$\Delta x = \dfrac{b-a}{n}$, 在每个小区间$[x_{k-1}, x_k]$上任取一点$\xi_k$, 则\textbf{黎曼和}定义为
    \[
        S_n = \sum_{k=1}^n f(\xi_k) \Delta x.
    \]
    如果当$n \to +\infty$时, 对于任意的$\xi_k$, 极限$\lim_{n \to +\infty} S_n$存在, 则称函数$f(x)$在$[a,b]$上\textbf{可积}, 并称此极限值为函数$f(x)$在$[a,b]$上的定积分, 记作
    \[
        \int_a^b f(x) dx = \lim_{n \to +\infty} S_n.
    \]
\end{definition}
\begin{note}
    % 分割的方法不唯一，可以等分，也可以不等分；选取的点$\xi_k$也不唯一，可以是区间的左端点、右端点或者任意一点.只要满足对于任意的$\xi_k$，极限存在即可.
    1.定积分的几何意义是:函数图像与积分坐标轴围成的\textbf{带符号面积}.

    2.将$(a,b)$特殊化为$(0,1)$,可以得到更简单的形式：
    \[
        \int_0^1 f(x)dx = \lim_{n \to +\infty} \sum_{k=1}^n f\left(\dfrac{k}{n}\right)\dfrac{1}{n} .
    \]
    $\bigstar\bigstar\bigstar$ 该式从右到左经常用来计算极限,
    这是常见的一类题目，重点是\ 提出“可爱因子”$\dfrac{1}{n}$并凑出积分变量(如$\dfrac{i}{n}$,或其他形式),根据积分变量的范围确定分范围.
    
\end{note}

\begin{induction}
    [定积分的性质]
    $\int_{a}^{a}f(x) dx = 0$, $\int_{b}^{a}f(x) dx = -\int_{a}^{b}f(x) dx$.
    
    (1)线性性质：$\int_a^b [\alpha f(x) + \beta g(x)] dx = \alpha \int_a^b f(x) dx + \beta \int_a^b g(x) dx$.

    (2)区间可加性：$\int_a^c f(x) dx = \int_a^b f(x) dx + \int_b^c f(x) dx$.

    (3)比较性质：如果$f(x) \le g(x)$对$[a,b]$内的任意$x$成立, 则
    \[
        \int_a^b f(x) dx \le \int_a^b g(x) dx.
    \]
    (4)\textbf{绝对值不等式}：$\left|\int_a^b f(x) dx\right| \le \int_a^b |f(x)| dx$.

    (5)估值定理：如果$m \le f(x) \le M$对$[a,b]$内的任意$x$成立, 则
    \[
        m(b-a) \le \int_a^b f(x) dx \le M(b-a).
    \]
    (6)$\bigstar\bigstar\bigstar$\textbf{积分中值定理}：如果函数$f(x)$在$[a,b]$上连续, 则存在$\xi\in[a,b]$使得
    \[
        \int_a^b f(x) dx = f(\xi)(b-a). 
    \]
    $\xi \in (a,b)$时，结论依旧成立.
\end{induction}

    在直角坐标系下曲线$y=f(x), x \in [a, b]$绕$x$轴旋转一周形成的旋转曲面的侧面积为
    $$S = \int_a^b 2\pi |y| ds = \int_a^b 2\pi |f(x)| \sqrt{1+[f'(x)]^2} dx$$
\end{induction}

\chapter{积分等式与积分不等式}
% ===================================
这一部分主要是使用前面学到的定理来解决有关积分的等式、不等式问题。

难度较大，做题的时候注意心态不要崩。


\begin{induction}
    [由中值定理推导的积分等式]
    设$f(x),g(x)$在$[a,b]$上连续，且$g(x)$不变号，则存在$\xi\in[a,b]$使得
    \[
        \int_a^b f(x)g(x)\mathrm{d}x = f(\xi)\int_a^b g(x)\mathrm{d}x.
    \]
\end{induction}
\begin{proof}
    设$g(x)\geqslant 0$，则存在$\xi_1,\xi_2\in[a,b]$使得
    \[
        f(\xi_1) = \min_{x\in[a,b]}f(x),\quad f(\xi_2) = \max_{x\in[a,b]}f(x).
    \]
    则
    \[
        f(\xi_1)\int_a^b g(x)\mathrm{d}x \leqslant \int_a^b f(x)g(x)\mathrm{d}x \leqslant f(\xi_2)\int_a^b g(x)\mathrm{d}x.
    \]
    由介值定理，存在$\xi\in[a,b]$使得
    \[
        \int_a^b f(x)g(x)\mathrm{d}x = f(\xi)\int_a^b g(x)\mathrm{d}x.
    \]
    当$g(x)\leqslant 0$时同理。
\end{proof}

\begin{induction}
    [使用夹逼准则求积分的极限]
    设$f(x),g(x),h(x)$在$[a,b]$上连续，
    且满足$f(x)\leqslant g(x)\leqslant h(x)$。

    若$\int_a^b f(x)\mathrm{d}x = \int_a^b h(x)\mathrm{d}x$，
    则$\lim\int_a^b g(x)\mathrm{d}x = \lim \int_a^b f(x)\mathrm{d}x$。
\end{induction}

\begin{induction}
    [$\bigstar$积分不等式的证明方法]
    (1)利用积分函数的单调性;
    
    (2)常量变量化：将某一个积分限变量化，构造变限积分函数，利用单调性证明该函数的单调性;
    
    (3)利用积分中值定理。    
    
    (4)利用泰勒公式将被积函数展开。
\end{induction}

\begin{example}
    [常量变量化方法]
    设$f(x)$在$[a,b]$上连续且$f(x)$严格单调增加，证明：
    $$(a+b)\int_{a}^{b}f(x)\mathrm{d}x < 2\int_{a}^{b}xf(x)\mathrm{d}x\mathstrut$$

    提示：将b变量化构造辅助函数$F(x)=(a+x)\int_{a}^{b}f(t)\mathrm{d}t - 2\int_{a}^{x}tf(t)\mathrm{d}t$
\end{example}

\begin{example}
    [积分中值定理方法]
    设$f'(x)$在$[a,b]$上连续$f(0)=0$，证明：
    $$|\int_{a}^{b}f(x)dx| \leq \dfrac{Ma^2}{2}$$
    其中$M=\max_{x\in[a,b]}|f'(x)|$。
\end{example}

\begin{example}
    [泰勒公式方法]
    设$f(x)$在$[0,1]$上二阶可导，且$f(\dfrac{1}{2})=1,f^{''}(x) > 0$，证明：
    $\int_{0}^{1}f(x)\mathrm{d}x\geq 1$
\end{example}
\begin{note}
    在$x_0=\dfrac{1}{2}$处泰勒展开，带佩亚诺余项，奇次项的积分为0.
\end{note}
